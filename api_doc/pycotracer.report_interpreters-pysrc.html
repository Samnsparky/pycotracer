 32 -def parse_yes_no_str(bool_str): 

 33      """Parse a string serialization of boolean data as yes (Y) or no (N). 
 34   
 35      Prase a string serialization of boolean data where True is "Y" and False is 
 36      "N" case-insensitive. 
 37   
 38      @param bool_str: The string to parse. 
 39      @type bool_str: str 
 40      @return: The interpreted string. 
 41      @rtype: bool 
 42      @raise ValueError: Raised if the passed string is not equal to 'N' or 'Y' 
 43          case insensitive. 
 44      """ 
 45      lower_bool_str = bool_str.lower() 
 46      if lower_bool_str == 'n': 
 47          return False 
 48      elif lower_bool_str == 'y': 
 49          return True 
 50      else: 
 51          raise ValueError('%s not a valid boolean string.' % bool_str)
